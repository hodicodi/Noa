DROP cascade DATABASE IF EXISTS noa;
CREATE DATABASE noa;

drop  TABLE IF EXISTS artists CASCADE;
DROP TABLE IF EXISTS users CASCADE;
DROP TABLE IF EXISTS songs CASCADE;


drop type if EXISTS artist_type_catagories CASCADE;
drop type if EXISTS genre_catagories CASCADE;
 

CREATE TYPE artist_type_catagories AS ENUM ('singer', 'band', 'pair');
CREATE TYPE genre_catagories AS ENUM ('rock', 'pop', 'hip-hop', 'country');


CREATE TABLE artists (
    uuid INT PRIMARY KEY,
    artist_type artist_type_catagories NOT NULL
);

CREATE TABLE users (
    uuid INT PRIMARY KEY,
    is_administor BOOLEAN  NOT null default false
);

CREATE TABLE songs (
    uuid INT PRIMARY KEY,
    artist INT references artists(uuid),
    publish_date DATE DEFAULT CURRENT_DATE, 
    genre genre_catagories NOT null
);
